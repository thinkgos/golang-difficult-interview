# 基本操作

```mysql
mysql -h $ip -P $port -u $user -p # 登录
show processlist;  # 查看连接状态
show variables like 'transaction_isolation'; # 查看所有变量

# 查找持续时间超过 60s 的事务
select * from information_schema.innodb_trx where TIME_TO_SEC(timediff(now(),trx_started))>60;

#  解释这条语句执行行为.
explain select * from t where between 10000 and 20000 limit 1000;
# 显示表索引
show index from t;
# 重新统计索引信息
analyze table t;

# 增加索引
alter table t add index idx_name(email);




# 设置innodb_io_capacity 这个参数了，它会告诉 InnoDB 你的磁盘能力
fio -filename=$filename -direct=1 -iodepth 1 -thread -rw=randrw -ioengine=psync -bs=16k -size=500M -numjobs=10 -runtime=10 -group_reporting -name=mytest 
```
